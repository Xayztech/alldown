Overview

AllDown API Documentation

A fast, free REST API to download media from YouTube, TikTok, Instagram, Twitter, Facebook, Scribd, and more. Supports multiple formats, quality selection, and search functionality.

✓
Free to use — No API key required
⚡
60 req/min — Rate limited per IP
🌐
CORS enabled — Use from any origin

Base URL

BASE URL 
https://xayz-downloader.vercel.app

All API endpoints are relative to this base URL. Replace xayz-downloader with your actual Vercel deployment URL.

Authentication

No authentication is required. All endpoints are public. Rate limiting is applied per IP address.

Note: For production use, consider adding an API key middleware to protect your deployment.

Rate Limiting

WindowMax RequestsScope
1 minute60 requestsPer IP address

When rate limited, the API returns 429 Too Many Requests with:

RESPONSE 429 
{ "success": false, "error": "Too many requests, please try again later." }

⬇ Downloader Endpoints

Universal Downloader (Auto-Detect)

Automatically detects the platform from the URL and routes to the correct handler.

GET /api/dl?url={url}
POST /api/dl body: { url }

Query Parameters

ParamTypeRequiredDescription
urlstring✓ YesAny supported platform URL
formatstringNomp4, mp3, webm, m4a, pdf (default: mp4)
qualitystringNo1080p, 720p, 480p, 360p, 240p, 144p, best (default: 720p)

Example Request

cURL 
curl "https://xayz-downloader.vercel.app/api/dl?url=https://youtu.be/dQw4w9WgXcQ&format=mp4&quality=720p"

💡 Code Examples

JavaScript / Node.js

JavaScript (fetch) 
const BASE_URL = 'https://xayz-downloader.vercel.app';

// Download TikTok video (no watermark)
async function downloadTikTok(url) {
  const res = await fetch(`${BASE_URL}/api/dl/tiktok?url=${encodeURIComponent(url)}`);
  const data = await res.json();
  
  if (!data.success) throw new Error(data.error);
  
  const { download, title, author } = data.data;
  console.log(`Title: ${title}`);
  console.log(`No watermark: ${download.no_watermark}`);
  console.log(`HD: ${download.hd}`);
  
  return data;
}

// Get YouTube formats then download
async function getYouTubeFormats(url) {
  const res = await fetch(`${BASE_URL}/api/dl/youtube/formats?url=${encodeURIComponent(url)}`);
  const data = await res.json();
  
  console.log('Available formats:');
  data.formats.forEach(f => {
    console.log(`  ${f.label} (${f.format}) - ${f.size || 'N/A'}`);
  });
  
  return data.formats;
}

// Search YouTube
async function searchYouTube(query, limit = 10) {
  const params = new URLSearchParams({ q: query, limit });
  const res = await fetch(`${BASE_URL}/api/search/youtube?${params}`);
  const data = await res.json();
  
  data.results.forEach(v => {
    console.log(`${v.title} by ${v.author} [${v.duration}]`);
    console.log(`  URL: ${v.url}`);
  });
  
  return data.results;
}

// Usage examples
await downloadTikTok('https://www.tiktok.com/@user/video/123');
await getYouTubeFormats('https://youtu.be/dQw4w9WgXcQ');
await searchYouTube('lofi music', 5);

Python

Python (requests) 
import requests

BASE_URL = 'https://xayz-downloader.vercel.app'
session = requests.Session()
session.headers.update({'User-Agent': 'MyApp/1.0'})

def download_youtube(url, format='mp4', quality='720p'):
    """Download YouTube video with specified format and quality"""
    res = session.get(f'{BASE_URL}/api/dl/youtube', params={
        'url': url, 'format': format, 'quality': quality
    })
    data = res.json()
    if not data['success']:
        raise Exception(data['error'])
    return data['data']

def download_tiktok(url):
    """Download TikTok video without watermark"""
    res = session.get(f'{BASE_URL}/api/dl/tiktok', params={'url': url})
    data = res.json()
    if not data['success']:
        raise Exception(data['error'])
    return data['data']

def download_scribd(url):
    """Download Scribd document"""
    res = session.get(f'{BASE_URL}/api/dl/scribd', params={'url': url})
    return res.json()

def search_youtube(query, limit=10):
    """Search YouTube videos"""
    res = session.get(f'{BASE_URL}/api/search/youtube', params={
        'q': query, 'limit': limit
    })
    return res.json()['results']

def detect_and_download(url):
    """Auto-detect platform and download"""
    res = session.get(f'{BASE_URL}/api/dl', params={'url': url})
    return res.json()

# ── Examples ─────────────────────────────────────────────
if __name__ == '__main__':
    # YouTube
    yt = download_youtube('https://youtu.be/dQw4w9WgXcQ', quality='1080p')
    print(f"YouTube: {yt['title']}")
    for fmt in yt['formats']:
        print(f"  {fmt['label']} -> {fmt['format']}")

    # TikTok
    tt = download_tiktok('https://www.tiktok.com/@user/video/123456')
    print(f"TikTok: {tt['title']}")
    print(f"  No watermark: {tt['download']['no_watermark']}")

    # Search
    results = search_youtube('python tutorial', limit=5)
    for v in results:
        print(f"  {v['title']} - {v['views']}")

PHP

PHP 
<?php
define('BASE_URL', 'https://xayz-downloader.vercel.app');

function apiRequest($endpoint, $params = []) {
    $url = BASE_URL . $endpoint;
    if (!empty($params)) {
        $url .= '?' . http_build_query($params);
    }
    
    $context = stream_context_create([
        'http' => [
            'timeout' => 15,
            'header' => 'User-Agent: MyApp/1.0'
        ]
    ]);
    
    $response = file_get_contents($url, false, $context);
    return json_decode($response, true);
}

// Download YouTube
function downloadYouTube($url, $format = 'mp4', $quality = '720p') {
    return apiRequest('/api/dl/youtube', [
        'url' => $url, 'format' => $format, 'quality' => $quality
    ]);
}

// Search YouTube
function searchYouTube($query, $limit = 10) {
    return apiRequest('/api/search/youtube', ['q' => $query, 'limit' => $limit]);
}

// Auto-detect download
function downloadAny($url) {
    return apiRequest('/api/dl', ['url' => $url]);
}

// Usage
$yt = downloadYouTube('https://youtu.be/dQw4w9WgXcQ');
echo $yt['data']['title'] . "\n";
foreach ($yt['data']['formats'] as $fmt) {
    echo "  {$fmt['label']} ({$fmt['format']})\n";
}
?>

Node.js with Axios

Node.js CJS 
'use strict';
const axios = require('axios');

const api = axios.create({
  baseURL: 'https://xayz-downloader.vercel.app',
  timeout: 15000
});

// Download any URL (auto-detect)
async function downloadAny(url, format = 'mp4', quality = '720p') {
  const { data } = await api.get('/api/dl', { params: { url, format, quality } });
  return data;
}

// Get YouTube info with all formats
async function getYouTubeFormats(url) {
  const { data } = await api.get('/api/dl/youtube/formats', { params: { url } });
  return data;
}

// Search YouTube
async function searchYouTube(q, limit = 10) {
  const { data } = await api.get('/api/search/youtube', { params: { q, limit } });
  return data.results;
}

// Download TikTok
async function downloadTikTok(url) {
  const { data } = await api.get('/api/dl/tiktok', { params: { url } });
  return data.data;
}

(async () => {
  // Download YouTube as MP3
  const audio = await downloadAny('https://youtu.be/dQw4w9WgXcQ', 'mp3', 'best');
  console.log('Audio:', audio.data.title);

  // Search
  const results = await searchYouTube('coding music', 3);
  results.forEach(v => console.log('-', v.title));
})();

🚀 Deploy Guide

Deploy your own instance of AllDown to Vercel in minutes.

1. Clone & Install

TERMINAL 
git clone https://github.com/yourusername/alldown-api.git
cd alldown-api
npm install

2. Run Locally

TERMINAL 
# Development mode (with auto-restart)
node --watch index.js

# Or standard
npm start

# API available at:
# http://localhost:3000/api
# http://localhost:3000/docs/api

3. Deploy to Vercel

TERMINAL 
# Install Vercel CLI
npm install -g vercel

# Login
vercel login

# Deploy
vercel

# Or deploy to production
vercel --prod

4. GitHub Auto-Deploy

TERMINAL 
git init
git add .
git commit -m "Initial commit: AllDown API"
git branch -M main
git remote add origin https://github.com/yourusername/alldown-api.git
git push -u origin main

# Then connect your GitHub repo to Vercel:
# vercel.com/new → Import Git Repository → Select repo
✓ Auto-Deploy: After connecting GitHub to Vercel, every push to main automatically deploys a new version.
